  
    def _build_heuristic_links(self, page_num: int):
        """
        [المنسق العام للشبكة]:
        المسؤول عن العمليات الحسابية الكبرى، المسح السيمانتيكي، وإدارة التوقيت.
        """
        self.logger.info("🕸️ البدء في نسج الروابط الاستدلالية الشاملة...")

        # 1. حساب التشابه السيمانتيكي الشامل (Global Semantic Scan)
        current_fingerprint = getattr(self, 'fingerprint', {})
        critical_zones = current_fingerprint.get("critical_heat_map", [])

        for i in range(len(self.vector_map)):
            # تحديد عمق المسح بناءً على أهمية المنطقة
            search_depth = 5 if (i + 1) in critical_zones else 2

            for j in range(i + 1, min(i + 1 + search_depth, len(self.vector_map))):
                similarity = self._calculate_semantic_similarity(i, j)

                if similarity > 0.75:
                    self.visual_links.append({
                        "source": i,
                        "target": j,
                        "type": "CRITICAL_CONNECTION",
                        "weight": round(float(similarity), 4),
                        "timestamp": self._get_pulse_timestamp()
                    })

        # 2. استدعاء "الوكيل" للتركيز على الصفحة الحالية وربطها بالماضي
        self._build_heuristic_links_agent(page_num)

        # 3. التنظيف الذكي (Smart Pruning) - صيانة الشبكة
        if len(self.visual_links) > 20000:
            self._prune_excessive_links()

    def _build_heuristic_links_agent(self, page_num: int):
        """
        [وكيل الربط الميداني]:
        يشرف على تأمين بيانات الصفحة، بناء روابط الـ DNA، واكتشاف مراكز الثقل (Hubs).
        """
        import collections

        # تأمين البيانات عبر البوابة السيادية
        page_data = self.get_page_data(page_num)
        if not page_data:
            self.logger.warning(f"⚠️ Agent: تعذر تأمين بيانات P{page_num+1}، إلغاء المهمة.")
            return

        # تهيئة البنية التحتية إذا لم تكن موجودة
        self._ensure_neural_infrastructure()

        keywords = set(page_data.get("semantic_keywords", []))
        current_layer = page_data.get("layer_type", "STANDARD_CONTENT")
        is_critical_zone = (page_num + 1) in range(60, 121)

        # الـ DNA الخاص بأوراقك البحثية
        ROBOTICS_DNA = {'jacobian', 'kinematics', 'manipulator', 'torque', 'actuator', 'control', 'velocity'}
        total_connections = 0

        # بناء روابط الكلمات المفتاحية
        for kw in keywords:
            kw_lower = kw.lower()
            is_dna = kw_lower in ROBOTICS_DNA
            term_importance = 1.8 if is_dna else 1.0

            # تسجيل الكلمة في الشبكة
            self.heuristic_network[kw].append({"page": page_num, "weight": 0.8 * term_importance})

            # عمق الذاكرة التاريخية
            history_depth = 100 if is_critical_zone else 50
            related_entries = self.heuristic_network[kw][-history_depth:]

            for entry in related_entries:
                past_page = entry["page"]
                if past_page == page_num: continue

                link_id = tuple(sorted((page_num, past_page)))
                if link_id not in self._link_registry:
                    self._link_registry.add(link_id)
                    total_connections += 1

                    # حساب القوة مع عقوبة المسافة
                    dist_penalty = 1.0 / (1.0 + abs(page_num - past_page) * 0.01)
                    strength = round(((0.8 * term_importance) + entry["weight"]) / 2 * dist_penalty, 3)

                    self.visual_links.append({
                        "origin": page_num, "target": past_page,
                        "strength": strength, "term": kw,
                        "type": "dna_bridge" if is_dna else "semantic_bridge",
                        "is_dna_link": is_dna
                    })

        # بناء الروابط المختصرة (Inference Shuttles) واكتشاف الـ Hubs
        self._manage_hub_connections(page_num, page_data, total_connections, is_critical_zone, current_layer)


  def _build_visual_heuristics(self, page_num: int, layout_data: Dict[str, Any]):
        """
        [المشرف البصري السيادي]: بناء شبكة الملاحة العصبية للمستند.
        """
        headings = layout_data.get("headings", [])

        # 1. تأمين المتغيرات الأساسية في الكلاس (يفضل أن تكون في __init__)
        if not hasattr(self, 'heading_map'):
            self.heading_map = defaultdict(list)
        if not hasattr(self, 'engine_metadata'):
            self.engine_metadata = {"dna_flow": []}

        # 2. معالجة العناوين
        for heading in headings:
            text_raw = heading.get("text", "")
            # تعريف title داخل الحلقة لضمان وجوده لكل عنوان
            title = str(text_raw).strip().lower()

            if len(title) < 3:
                continue

            # تسجيل العنوان
            self.heading_map[title].append(page_num)

            # الاستمرارية الدلالية
            if len(self.heading_map[title]) > 1:
                source_page = self.heading_map[title][-2]
                is_high_priority = heading.get("priority") == "HIGH"

                self.visual_links.append({
                    "link_type": "HEADING_CONTINUITY",
                    "origin": source_page,
                    "target": page_num,
                    "anchor": title,
                    "strength": 2.0 if is_high_priority else 1.4
                })

                # تسجيل الـ Hub الإستراتيجي (title متاح هنا بالتأكيد)
                hub_id = f"v_hub_{title.replace(' ', '_')}"
                if not hasattr(self, hub_id):
                    self.network_hubs_count += 1
                    setattr(self, hub_id, True)
                    self.engine_metadata["dna_flow"].append(f"Hub detected: {title}")

        # 3. التدفق الهيكلي (خارج حلقة العناوين لأنها تخص الصفحة ككل)
        if page_num > 0:
            self.visual_links.append({
                "link_type": "STRUCTURAL_FLOW",
                "origin": page_num - 1,
                "target": page_num,
                "strength": 0.9
            })

        self.logger.info(f"🕸️ Visual Grid: P{page_num} processed. Hubs: {self.network_hubs_count}")
